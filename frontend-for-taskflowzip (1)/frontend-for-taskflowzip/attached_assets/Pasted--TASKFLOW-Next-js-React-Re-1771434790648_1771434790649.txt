## 🎯 КЛЮЧЕВЫЕ ТРЕБОВАНИЯ

### 1.1 Интеграция с API бэкенда

```typescript
// GET /api/v1/projects/{projectSlug}/tasks/graph
// Возвращает:
{
  nodes: Array<{
    id: string;              // string, совпадает с task.id
    type: 'taskNode';
    data: {
      id: number;            // task.id для запросов
      name: string;
      status: 'todo' | 'in_progress' | 'review' | 'completed' | 'blocked';
      status_color: string;  // HEX из бэкенда
      assignee: string | null;
      creator: string;
      priority: 0 | 1 | 2;
      deadline: string | null;
      is_ready: boolean;     // true если задача готова к выполнению
    };
    position: { x: number; y: number };
  }>;
  edges: Array<{
    id: string;              // `e${source}-${target}`
    source: string;
    target: string;
    type: 'dependency';
    data: {
      actions: Array<{
        type: string;        // 'notify_assignee', 'notify_creator', 'change_status'
        delay: number;
      }>;
    };
    animated: boolean;       // true для анимации
    label?: string;
  }>;
}
```

## 🎮 БИЗНЕС-ЛОГИКА ВЗАИМОДЕЙСТВИЯ

### 3.1 Создание/удаление ребер

```typescript
// 1. Пользователь тянет от source к target
onConnect: (connection) => {
  // Optimistic update
  addEdgeOptimistically(connection);
  
  // API запрос
  fetch(`/api/v1/projects/${slug}/tasks/${sourceId}/dependencies`, {
    method: 'POST',
    body: JSON.stringify({ target_task_id: targetId })
  }).catch(() => {
    // Откат при ошибке
    removeEdge(connection);
    showError('Циклическая зависимость запрещена');
  });
}

// 2. Удаление ребра
onEdgeDelete: (edgeId) => {
  deleteEdgeOptimistically(edgeId);
  fetch(`/api/v1/projects/${slug}/tasks/dependencies/${edgeId}`, {
    method: 'DELETE'
  }).catch(() => revertEdge(edgeId));
}
```

### 3.3 Изменение статуса задачи

```typescript
const changeStatus = async (taskId: number, newStatus: TaskStatus) => {
  // 1. Мгновенное обновление UI
  updateNodeOptimistically(taskId, { status: newStatus });
  
  // 2. API запрос
  const response = await fetch(
    `/api/v1/projects/${slug}/tasks/${taskId}/status`,
    {
      method: 'POST',
      body: JSON.stringify({ status: newStatus })
    }
  );
  
  if (!response.ok) {
    // Откат при ошибке
    revertNode(taskId);
    return;
  }
  
  const data = await response.json();
  
  // 3. Показать выполненные действия
  if (data.actions_executed.length > 0) {
    showNotification(`Выполнено действий: ${data.actions_executed.length}`);
  }
  
  // 4. Обновить is_ready для зависимых задач
  refreshReadinessForDownstream(taskId);
};
```

## ⚡ ОПТИМИЗАЦИЯ ПРОИЗВОДИТЕЛЬНОСТИ

### 5.1 Кэширование

```typescript
// React Query configuration
const graphQuery = useQuery({
  queryKey: ['graph', projectSlug],
  queryFn: fetchGraph,
  staleTime: 30 * 1000,        // 30 секунд
  cacheTime: 5 * 60 * 1000,     // 5 минут
});

// Optimistic updates
const addDependency = useMutation({
  mutationFn: createDependency,
  onMutate: async (newEdge) => {
    await queryClient.cancelQueries(['graph', projectSlug]);
    const previousGraph = queryClient.getQueryData(['graph', projectSlug]);
    
    queryClient.setQueryData(['graph', projectSlug], (old) => ({
      ...old,
      edges: [...old.edges, newEdge]
    }));
    
    return { previousGraph };
  },
  onError: (err, newEdge, context) => {
    queryClient.setQueryData(['graph', projectSlug], context.previousGraph);
  }
});
```

### 5.3 Debounce для автосохранения

```typescript
// Сохраняем позиции узлов
const debouncedSave = useMemo(
  () => debounce((nodes) => {
    fetch(`/api/v1/projects/${slug}/tasks/graph`, {
      method: 'PUT',
      body: JSON.stringify({ nodes, edges: currentEdges })
    });
  }, 500),
  [slug, edges]
);

const onNodesChange = useCallback((changes) => {
  setNodes((nds) => applyNodeChanges(changes, nds));
  
  // Сохраняем только позиции, не при каждом пикселе!
  const hasPositionChange = changes.some(c => c.type === 'position');
  if (hasPositionChange) {
    debouncedSave(nodes);
  }
}, [nodes, debouncedSave]);
```
